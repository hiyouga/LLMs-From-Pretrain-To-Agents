\section{Next Frontiers in LLM and Agent Systems}

\begin{frame}{Long-Horizon Reinforcement Learning}
    \begin{itemize}
        \item \textbf{Longer reasoning traces}
            \begin{itemize}
                \item Models will execute deeper multi-step thought processes to solve highly non-trivial problems.
            \end{itemize}
        \item \textbf{Extended interaction rounds}
            \begin{itemize}
                \item Agents will maintain coherence across much longer sequences of feedback and environment responses.
            \end{itemize}
        \item \textbf{Complex agent environments}
            \begin{itemize}
                \item Systems will evolve to navigate high-dimensional and unpredictable task spaces with minimal supervision.
            \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Broader Agent Environments}
    \begin{itemize}
        \item \textbf{Professional softwares}
            \begin{itemize}
                \item Future agents will seamlessly interface with a vast array of specialized professional software tools.
            \end{itemize}
        \item \textbf{Operating systems}
            \begin{itemize}
                \item Agents will gain the capacity to operate directly on operating system interfaces and file systems.
            \end{itemize}
        \item \textbf{Edge devices}
            \begin{itemize}
                \item Agents will extend their operational environments to physical edge devices such as robotic systems.
            \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{World Models}
    \begin{itemize}
        \item \textbf{Physical world modeling}
            \begin{itemize}
                \item Research will focus on teaching models to internalize the causal laws and constraints of the physical universe.
            \end{itemize}
        \item \textbf{Advanced video generation}
            \begin{itemize}
                \item Generative models will move toward maintaining perfect temporal consistency and object permanence in video.
            \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Continual Learning and Adaptation}
    \begin{itemize}
        \item \textbf{Nested learning architecture}
            \begin{itemize}
                \item Google's nested learning framework utilizes varying parameter update frequencies to alleviate catastrophic forgetting.
            \end{itemize}
        \item \textbf{Cross-scenario adaptation}
            \begin{itemize}
                \item Models will autonomously fine-tune their internal states to excel across rapidly changing task domains.
            \end{itemize}
    \end{itemize}
\end{frame}
